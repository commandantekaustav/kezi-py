# This function determines the recommendation based on criteria.
def get_recommendation(score, has_required_skill):
    """Generates a hiring recommendation."""
    if score >= 90 and has_required_skill:
        return "STRONG HIRE - Priority candidate."
    elif score >= 75 and has_required_skill:
        return "HIRE - Meets all qualifications."
    else:
        return "REJECT - Does not meet minimum criteria."

# --- Main Program Starts Here ---
print("--- Automated Candidate Screening Protocol ---")

# 1. Get input from the user
name = input("Enter candidate's name: ")
score_str = input(f"Enter {name}'s interview score (0-100): ")
skill_response = input("Does the candidate have the core skill? (yes/no): ")

# 2. Process and clean the data
score = 0
try:
    score = int(score_str) # Cast score to an integer
except:
    print("Invalid score entered. Defaulting to 0.")

# Convert 'yes' to a boolean True, anything else to False
has_skill = skill_response.lower().strip() == 'yes' 

# 3. Use the function to make a decision
recommendation = get_recommendation(score, has_skill)
print(f"\nRecommendation for {name}: {recommendation}\n")

# 4. Write the result to a log file
with open('screening_log.txt', 'a') as f:
    log_entry = f"Candidate: {name}, Score: {score}, Recommendation: {recommendation}\n"
    f.write(log_entry)

print("Screening complete. Result saved to log.")
